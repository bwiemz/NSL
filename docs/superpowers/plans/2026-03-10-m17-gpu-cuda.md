# M17: GPU/CUDA + `kernel` Keyword Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Enable end-to-end GPU training and custom kernel execution in NSL via CUDA.

**Architecture:** Unified `NslTensor` struct extended with `device` and `dtype` fields. CUDA support via `cudarc` crate (driver API only) as an optional cargo feature. User-defined kernels compile to PTX strings embedded in `.rodata`, launched via `cuModuleLoadData` JIT. Standard tensor ops dispatch to GPU PTX kernels stored as `const &str` in the runtime. Vertical slice approach — prove the full pipeline end-to-end with a trivial kernel before widening to all ops.

**Tech Stack:** Rust, Cranelift (AOT codegen), `cudarc` 0.12 (CUDA driver API), PTX ISA 7.0, CUDA Unified Memory

**Spec:** `docs/plans/2026-03-10-m17-gpu-cuda-design.md`

```rust
//! CUDA runtime: context management, kernel launch, module cache.
//! Only compiled when the `cuda` feature is enabled.

#[cfg(feature = "cuda")]
mod inner {
    use cudarc::driver::sys::*;
    use std::sync::{Mutex, OnceLock};
    use std::collections::HashMap;
    use std::ffi::CString;

    /// Global CUDA context, lazily initialized.
    struct CudaState {
        device: CUdevice,
        context: CUcontext,
        /// Cache: PTX pointer address → compiled CUmodule
        module_cache: HashMap<usize, CUmodule>,
    }

    static CUDA_STATE: OnceLock<Mutex<CudaState>> = OnceLock::new();

    fn ensure_init() -> &'static Mutex<CudaState> {
        CUDA_STATE.get_or_init(|| {
            unsafe {
                let mut result = cuInit(0);
                assert_eq!(result, CUresult::CUDA_SUCCESS, "cuInit failed: {:?}", result);

                let mut device: CUdevice = 0;
                result = cuDeviceGet(&mut device, 0);
                assert_eq!(result, CUresult::CUDA_SUCCESS, "cuDeviceGet failed: {:?}", result);

                let mut context: CUcontext = std::ptr::null_mut();
                result = cuCtxCreate_v2(&mut context, 0, device);
                assert_eq!(result, CUresult::CUDA_SUCCESS, "cuCtxCreate failed: {:?}", result);

                Mutex::new(CudaState {
                    device,
                    context,
                    module_cache: HashMap::new(),
                })
            }
        })
    }
}

/// FFI: Initialize CUDA. Called by codegen at program start if CUDA tensors detected.
#[no_mangle]
pub extern "C" fn nsl_cuda_init() -> i64 {
    #[cfg(feature = "cuda")]
    {
        inner::ensure_init();
        0 // success
    }
    #[cfg(not(feature = "cuda"))]
    {
        eprintln!("CUDA support not compiled. Rebuild with --features cuda");
        std::process::abort();
    }
}
```

```rust
    /// Allocate unified memory (accessible from both CPU and GPU).
    pub(crate) fn alloc_managed(size_bytes: usize) -> *mut std::ffi::c_void {
        ensure_init();
        unsafe {
            let mut ptr: CUdeviceptr = 0;
            let result = cuMemAllocManaged(
                &mut ptr,
                size_bytes,
                CUmemAttach_flags::CU_MEM_ATTACH_GLOBAL as u32,
            );
            assert_eq!(result, CUresult::CUDA_SUCCESS,
                "cuMemAllocManaged({} bytes) failed: {:?}", size_bytes, result);
            ptr as *mut std::ffi::c_void
        }
    }

    /// Free unified memory.
    pub(crate) fn free_managed(ptr: *mut std::ffi::c_void) {
        unsafe {
            let result = cuMemFree_v2(ptr as CUdeviceptr);
            assert_eq!(result, CUresult::CUDA_SUCCESS,
                "cuMemFree failed: {:?}", result);
        }
    }

    /// Prefetch memory to a specific device. Best-effort: silently ignores
    /// CUDA_ERROR_NOT_SUPPORTED on devices without concurrent managed access.
    pub(crate) fn prefetch_to_device(ptr: *mut std::ffi::c_void, size_bytes: usize, device_id: i32) {
        unsafe {
            let result = cuMemPrefetchAsync(
                ptr as CUdeviceptr,
                size_bytes,
                device_id,
                std::ptr::null_mut(), // default stream (stream 0)
            );
            // Silently ignore NOT_SUPPORTED — driver handles page migration on demand
            if result != CUresult::CUDA_SUCCESS
                && result != CUresult::CUDA_ERROR_NOT_SUPPORTED
            {
                panic!("cuMemPrefetchAsync failed: {:?}", result);
            }
        }
    }
```

```rust
    /// Launch a PTX kernel on the GPU.
    /// ptx_ptr: null-terminated PTX string
    /// name_ptr: null-terminated kernel entry point name
    pub(crate) fn kernel_launch(
        ptx_ptr: *const u8,
        name_ptr: *const u8,
        grid: [i64; 3],
        block: [i64; 3],
        args: &[*mut std::ffi::c_void],
    ) -> CUresult {
        let state = ensure_init();
        let mut guard = state.lock().unwrap();

        // Cache lookup by PTX pointer address
        let cache_key = ptx_ptr as usize;
        let module = if let Some(m) = guard.module_cache.get(&cache_key) {
            *m
        } else {
            // JIT compile the PTX
            let mut module: CUmodule = std::ptr::null_mut();
            unsafe {
                let result = cuModuleLoadData(&mut module, ptx_ptr as *const std::ffi::c_void);
                if result != CUresult::CUDA_SUCCESS {
                    eprintln!("cuModuleLoadData failed: {:?}", result);
                    return result;
                }
            }
            guard.module_cache.insert(cache_key, module);
            module
        };

        // Get function handle
        let mut func: CUfunction = std::ptr::null_mut();
        unsafe {
            let result = cuModuleGetFunction(&mut func, module, name_ptr as *const i8);
            if result != CUresult::CUDA_SUCCESS {
                eprintln!("cuModuleGetFunction failed: {:?}", result);
                return result;
            }
        }

        // Build args array for cuLaunchKernel
        let mut kernel_args: Vec<*mut std::ffi::c_void> = args
            .iter()
            .map(|a| a as *const _ as *mut std::ffi::c_void)
            .collect();

        unsafe {
            cuLaunchKernel(
                func,
                grid[0] as u32, grid[1] as u32, grid[2] as u32,
                block[0] as u32, block[1] as u32, block[2] as u32,
                0,                          // shared memory bytes
                std::ptr::null_mut(),       // stream (default)
                kernel_args.as_mut_ptr(),
                std::ptr::null_mut(),       // extra
            )
        }
    }
```

```rust
/// Transfer a tensor to a different device.
/// CPU→GPU: converts f64 → f32, allocates unified memory, prefetches.
/// GPU→CPU: converts f32 → f64, allocates CPU memory.
#[no_mangle]
pub extern "C" fn nsl_tensor_to_device(tensor_ptr: i64, target_device: i64) -> i64 {
    let t = unsafe { &*(tensor_ptr as *const NslTensor) };
    let target = target_device as u8;

    if t.device == target {
        // Same device — increment refcount and return same pointer
        let t_mut = unsafe { &mut *(tensor_ptr as *mut NslTensor) };
        t_mut.refcount += 1;
        return tensor_ptr;
    }

    let len = t.len as usize;

    if t.device == 0 && target > 0 {
        // CPU → GPU: f64 → f32
        #[cfg(feature = "cuda")]
        {
            let src = t.data_f64();
            let dst_size = len * std::mem::size_of::<f32>();
            let dst = crate::cuda::inner::alloc_managed(dst_size);
            let dst_f32 = dst as *mut f32;

            // Convert f64 → f32 on CPU side (unified memory)
            for i in 0..len {
                unsafe { *dst_f32.add(i) = *src.add(i) as f32; }
            }

            // Prefetch to GPU
            crate::cuda::inner::prefetch_to_device(dst, dst_size, (target - 1) as i32);

            // Copy shape and strides
            let shape = NslTensor::copy_shape(t.shape, t.ndim);
            let strides = NslTensor::compute_strides(shape, t.ndim);

            let new_t = Box::new(NslTensor {
                data: dst,
                shape,
                strides,
                ndim: t.ndim,
                len: t.len,
                refcount: 1,
                device: target,
                dtype: 1, // f32
            });
            return Box::into_raw(new_t) as i64;
        }
        #[cfg(not(feature = "cuda"))]
        { panic!("CUDA support not compiled"); }
    }

    if t.device > 0 && target == 0 {
        // GPU → CPU: f32 → f64
        #[cfg(feature = "cuda")]
        {
            // Sync to ensure GPU writes are visible
            unsafe { cudarc::driver::sys::cuCtxSynchronize(); }

            let src = t.data_f32();
            let dst_size = len * std::mem::size_of::<f64>();
            let layout = std::alloc::Layout::from_size_align(dst_size, 8).unwrap();
            let dst = unsafe { std::alloc::alloc(layout) as *mut f64 };
            let dst_void = dst as *mut c_void;

            // Convert f32 → f64
            for i in 0..len {
                unsafe { *dst.add(i) = *src.add(i) as f64; }
            }

            let shape = NslTensor::copy_shape(t.shape, t.ndim);
            let strides = NslTensor::compute_strides(shape, t.ndim);

            let new_t = Box::new(NslTensor {
                data: dst_void,
                shape,
                strides,
                ndim: t.ndim,
                len: t.len,
                refcount: 1,
                device: 0,
                dtype: 0, // f64
            });
            return Box::into_raw(new_t) as i64;
        }
        #[cfg(not(feature = "cuda"))]
        { panic!("CUDA support not compiled"); }
    }

    panic!("GPU-to-GPU transfer not yet supported");
}
```

```rust
//! KernelCompiler: translates KernelDef AST nodes into PTX strings.
//!
//! For M17, this generates PTX from the kernel body using a direct
//! AST-to-PTX translation. PTX virtual registers are used (unlimited),
//! and the NVIDIA driver's ptxas handles physical register allocation.

use crate::compiler::Compiler;
use nsl_ast::block::KernelDef;

pub struct KernelCompiler {
    ptx: String,
    next_u32: u32,        // .u32 register counter
    next_u64: u32,        // .u64 register counter
    next_f32: u32,        // .f32 register counter
    next_pred: u32,       // .pred register counter
    // Register maps for named variables
    float_regs: std::collections::HashMap<String, String>,
    int_regs: std::collections::HashMap<String, String>,
}

impl KernelCompiler {
    pub fn new() -> Self {
        Self {
            ptx: String::new(),
            next_u32: 0,
            next_u64: 0,
            next_f32: 0,
            next_pred: 0,
            float_regs: std::collections::HashMap::new(),
            int_regs: std::collections::HashMap::new(),
        }
    }

    /// Compile a KernelDef into a null-terminated PTX string.
    pub fn compile(kernel: &KernelDef, interner: &nsl_ast::Interner) -> Vec<u8> {
        let mut kc = Self::new();
        let name = interner.resolve(kernel.name).unwrap();

        // PTX header
        kc.ptx.push_str(".version 7.0\n");
        kc.ptx.push_str(".target sm_52\n");
        kc.ptx.push_str(".address_size 64\n\n");

        // Entry point with parameters
        kc.ptx.push_str(&format!(".visible .entry {}(\n", name));
        for (i, param) in kernel.params.iter().enumerate() {
            let pname = interner.resolve(param.name).unwrap();
            // All tensor params passed as u64 pointers
            kc.ptx.push_str(&format!("    .param .u64 {}", pname));
            if i < kernel.params.len() - 1 {
                kc.ptx.push(',');
            }
            kc.ptx.push('\n');
        }
        kc.ptx.push_str(") {\n");

        // Register declarations (filled in during body compilation)
        let reg_pos = kc.ptx.len();
        kc.ptx.push_str("    // REGISTERS_PLACEHOLDER\n");

        // Load parameters
        for param in &kernel.params {
            let pname = interner.resolve(param.name).unwrap();
            let rd = kc.alloc_u64_reg();
            kc.ptx.push_str(&format!("    ld.param.u64 {}, [{}];\n", rd, pname));
            kc.int_regs.insert(pname.to_string(), rd);
        }

        // Compile body statements
        kc.compile_block(&kernel.body, interner);

        kc.ptx.push_str("DONE:\n    ret;\n}\n");

        // Replace register placeholder with actual declarations
        let reg_decls = kc.generate_reg_declarations();
        kc.ptx = kc.ptx.replace("    // REGISTERS_PLACEHOLDER\n", &reg_decls);

        // Null-terminate for CUDA driver API
        let mut bytes = kc.ptx.into_bytes();
        bytes.push(0);
        bytes
    }

    fn alloc_u64_reg(&mut self) -> String {
        let r = format!("%rd{}", self.next_u64);
        self.next_u64 += 1;
        r
    }

    fn alloc_f32_reg(&mut self) -> String {
        let r = format!("%fs{}", self.next_f32);
        self.next_f32 += 1;
        r
    }

    fn alloc_u32_reg(&mut self) -> String {
        let r = format!("%r{}", self.next_u32);
        self.next_u32 += 1;
        r
    }

    fn alloc_pred_reg(&mut self) -> String {
        let r = format!("%p{}", self.next_pred);
        self.next_pred += 1;
        r
    }

    fn generate_reg_declarations(&self) -> String {
        // Emit declarations only for register types actually used
        let mut decls = String::new();
        if self.next_u32 > 0 {
            decls.push_str(&format!("    .reg .u32 %r<{}>;\n", self.next_u32));
        }
        if self.next_u64 > 0 {
            decls.push_str(&format!("    .reg .u64 %rd<{}>;\n", self.next_u64));
        }
        if self.next_f32 > 0 {
            decls.push_str(&format!("    .reg .f32 %fs<{}>;\n", self.next_f32));
        }
        if self.next_pred > 0 {
            decls.push_str(&format!("    .reg .pred %p<{}>;\n", self.next_pred));
        }
        decls
    }

    fn compile_block(&mut self, block: &nsl_ast::block::Block, interner: &nsl_ast::Interner) {
        for stmt in &block.stmts {
            self.compile_stmt(stmt, interner);
        }
    }

    fn compile_stmt(&mut self, stmt: &nsl_ast::stmt::Stmt, interner: &nsl_ast::Interner) {
        // Handle: let assignments, if/else, while, expression statements
        // Map intrinsics: thread_id() → mov.u32 %rN, %tid.x
        // Map tensor index: a[idx] → compute address, ld.global.f32
        // Map arithmetic: +,-,*,/ → add.f32, sub.f32, mul.f32, div.f32
        // Map comparisons: < → setp.lt, > → setp.gt, etc.
        // Map if/else → @pred bra / bra labels
        todo!("Implement AST-to-PTX statement compilation")
    }
}
```

**Note:** The full `compile_stmt` and `compile_expr` implementations are the bulk of this task. They handle:
- `let idx = thread_id()` → `mov.u32 %rN, %tid.x; cvt.u64.u32 %rdN, %rN`
- `a[idx]` → address calculation + `ld.global.f32`

- Arithmetic → `add.f32`, `mul.f32`, etc.

```rust
//! Semantic validation for `kernel` blocks.

use nsl_ast::stmt::{Stmt, StmtKind};
use nsl_ast::expr::{Expr, ExprKind};
use crate::types::Device;

/// Set of allowed intrinsic names inside kernel bodies.
const KERNEL_INTRINSICS: &[&str] = &[
    "thread_id", "thread_id_y",
    "block_id", "block_id_y",
    "block_dim",
    "sync_threads",
    "atomic_add",
];

/// Validate that a kernel body only contains allowed operations.
/// Returns a list of error messages for violations.
pub fn validate_kernel_body(body: &[Stmt], interner: &nsl_ast::Interner) -> Vec<String> {
    let mut errors = Vec::new();

    for stmt in body {
        validate_stmt(stmt, interner, &mut errors);
    }

    errors
}

fn validate_stmt(stmt: &Stmt, interner: &nsl_ast::Interner, errors: &mut Vec<String>) {
    match &stmt.kind {
        StmtKind::Let { .. } => {} // Allowed
        StmtKind::Expr(expr) => validate_expr(expr, interner, errors),
        StmtKind::If { .. } => {} // Allowed — recurse into branches
        StmtKind::While { .. } => {} // Allowed
        StmtKind::Assign { .. } => {} // Allowed (tensor element assignment)
        StmtKind::SharedDecl { .. } => {} // Allowed
        // Disallowed:
        StmtKind::FnDef(_) => errors.push("function definitions not allowed in kernel body".into()),
        StmtKind::ModelDef(_) => errors.push("model definitions not allowed in kernel body".into()),
        StmtKind::Print(_) => errors.push("print not allowed in kernel body".into()),
        StmtKind::Import(_) => errors.push("imports not allowed in kernel body".into()),
        _ => errors.push(format!("statement type {:?} not allowed in kernel body", stmt.kind)),
    }
}

fn validate_expr(expr: &Expr, interner: &nsl_ast::Interner, errors: &mut Vec<String>) {
    match &expr.kind {
        ExprKind::Call { func, .. } => {
            // Check if it's a kernel intrinsic
            if let ExprKind::Name(name) = &func.kind {
                let name_str = interner.resolve(*name).unwrap();
                if !KERNEL_INTRINSICS.contains(&name_str) {
                    errors.push(format!(
                        "function '{}' is not a GPU intrinsic and cannot be called in a kernel body",
                        name_str
                    ));
                }
            }
        }
        // Arithmetic, comparisons, indexing — all allowed
        _ => {}
    }
}
```
