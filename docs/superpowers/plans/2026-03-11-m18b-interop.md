# M18b Interop Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Enable NeuralScript to load HuggingFace pretrained weights, serialize in safetensors format, and export ONNX inference graphs — all in pure Rust.

**Architecture:** Feature-gated interop crate extensions (`safetensors`, `hf-hub`, `prost`, `half`) behind `#[cfg(feature = "interop")]`. Safetensors provides the weight I/O foundation. HF loading uses compiler-generated offset tables (no runtime reflection). ONNX export uses trace-based capture with pointer-lifecycle auto-resolution of weights.

**Tech Stack:** Rust (Cranelift codegen), `safetensors 0.4`, `hf-hub 0.3`, `prost 0.13`, `half 2.4`

**Design spec:** `docs/superpowers/specs/2026-03-11-m18b-interop-design.md`

```rust
use std::ffi::c_void;

use crate::dict;
use crate::list;
use crate::memory;
use crate::string;
use crate::tensor::NslTensor;

/// Load a safetensors file and return an NslDict mapping param names to NslTensors.
/// All tensors are loaded as f32 (dtype=1) regardless of source dtype.
/// device: 0 = CPU, 1+ = GPU (requires CUDA feature).
#[no_mangle]
pub extern "C" fn nsl_safetensors_load(path_ptr: i64, path_len: i64, device: i64) -> i64 {
    let path = unsafe {
        let slice = std::slice::from_raw_parts(path_ptr as *const u8, path_len as usize);
        std::str::from_utf8_unchecked(slice)
    };

    // Read file bytes (mmap for large files, read for small)
    let file_bytes = match std::fs::read(path) {
        Ok(b) => b,
        Err(e) => {
            eprintln!("[nsl] Error reading safetensors file '{}': {}", path, e);
            std::process::abort();
        }
    };

    let st = match safetensors::SafeTensors::deserialize(&file_bytes) {
        Ok(st) => st,
        Err(e) => {
            eprintln!("[nsl] Error parsing safetensors file '{}': {}", path, e);
            std::process::abort();
        }
    };

    let dict_ptr = dict::nsl_dict_new();

    for (name, tensor_view) in st.tensors() {
        let shape_slice = tensor_view.shape();
        let ndim = shape_slice.len() as i64;
        let len: i64 = shape_slice.iter().map(|&d| d as i64).product();
        if len == 0 {
            continue; // skip empty tensors
        }

        // Convert source data to f32
        let f32_data = convert_to_f32(tensor_view.dtype(), tensor_view.data());

        // Allocate NslTensor as f32 on target device
        let tensor = allocate_f32_tensor(&f32_data, shape_slice, ndim, len, device as u8);
        let tensor_ptr = Box::into_raw(Box::new(tensor)) as i64;

        // Allocate a null-terminated C string for the key
        let key_ptr = alloc_c_string(&name);

        dict::nsl_dict_set_str(dict_ptr, key_ptr as i64, tensor_ptr);
    }

    dict_ptr
}

/// Convert raw bytes from any safetensors dtype to Vec<f32>.
fn convert_to_f32(dtype: safetensors::Dtype, data: &[u8]) -> Vec<f32> {
    match dtype {
        safetensors::Dtype::F32 => {
            bytemuck::cast_slice(data).to_vec()
        }
        safetensors::Dtype::F16 => {
            let f16_slice: &[half::f16] = bytemuck::cast_slice(data);
            f16_slice.iter().map(|v| v.to_f32()).collect()
        }
        safetensors::Dtype::BF16 => {
            let bf16_slice: &[half::bf16] = bytemuck::cast_slice(data);
            bf16_slice.iter().map(|v| v.to_f32()).collect()
        }
        safetensors::Dtype::F64 => {
            let f64_slice: &[f64] = bytemuck::cast_slice(data);
            f64_slice.iter().map(|&v| v as f32).collect()
        }
        other => {
            eprintln!("[nsl] Unsupported safetensors dtype: {:?}", other);
            std::process::abort();
        }
    }
}

/// Allocate an NslTensor filled with f32 data.
fn allocate_f32_tensor(
    f32_data: &[f32],
    shape_slice: &[usize],
    ndim: i64,
    len: i64,
    device: u8,
) -> NslTensor {
    let data_bytes = f32_data.len() * std::mem::size_of::<f32>();

    let data_ptr: *mut c_void = if device == 0 {
        // CPU allocation
        let ptr = unsafe { memory::checked_alloc(data_bytes) } as *mut f32;
        unsafe {
            std::ptr::copy_nonoverlapping(f32_data.as_ptr(), ptr, f32_data.len());
        }
        ptr as *mut c_void
    } else {
        // GPU allocation via CUDA unified memory
        #[cfg(feature = "cuda")]
        {
            let ptr = crate::cuda::alloc_managed(data_bytes);
            unsafe {
                std::ptr::copy_nonoverlapping(
                    f32_data.as_ptr() as *const u8,
                    ptr as *mut u8,
                    data_bytes,
                );
            }
            ptr as *mut c_void
        }
        #[cfg(not(feature = "cuda"))]
        {
            eprintln!("[nsl] GPU device requested but CUDA feature not enabled");
            std::process::abort();
        }
    };

    // Allocate shape array
    let shape_ptr = unsafe { memory::checked_alloc(ndim as usize * 8) } as *mut i64;
    for (i, &dim) in shape_slice.iter().enumerate() {
        unsafe { *shape_ptr.add(i) = dim as i64; }
    }

    let strides_ptr = NslTensor::compute_strides(shape_ptr, ndim);

    NslTensor {
        data: data_ptr,
        shape: shape_ptr,
        strides: strides_ptr,
        ndim,
        len,
        refcount: 1,
        device,
        dtype: 1, // always f32
    }
}

/// Allocate a null-terminated C string on the heap, return pointer.
fn alloc_c_string(s: &str) -> *mut u8 {
    let bytes = s.as_bytes();
    let ptr = unsafe { memory::checked_alloc(bytes.len() + 1) };
    unsafe {
        std::ptr::copy_nonoverlapping(bytes.as_ptr(), ptr, bytes.len());
        *ptr.add(bytes.len()) = 0; // null terminator
    }
    ptr
}
```

```rust
/// Save an NslDict of tensors to a safetensors file.
/// All tensors are saved as f32 regardless of source dtype.
#[no_mangle]
pub extern "C" fn nsl_safetensors_save(dict_ptr: i64, path_ptr: i64, path_len: i64) {
    let path = unsafe {
        let slice = std::slice::from_raw_parts(path_ptr as *const u8, path_len as usize);
        std::str::from_utf8_unchecked(slice)
    };

    let dict = crate::dict::NslDict::from_ptr(dict_ptr);

    // Collect all key-value pairs from the dict
    let mut tensors_data: Vec<(String, Vec<f32>, Vec<usize>)> = Vec::new();

    let keys_list_ptr = dict::nsl_dict_keys(dict_ptr);
    let keys_list = crate::list::NslList::from_ptr(keys_list_ptr);

    for i in 0..keys_list.len as usize {
        let key_ptr = unsafe { *keys_list.data.add(i) };
        let key_cstr = unsafe { std::ffi::CStr::from_ptr(key_ptr as *const i8) };
        let key_str = key_cstr.to_str().unwrap_or("unknown").to_string();

        let tensor_ptr = dict::nsl_dict_get_str(dict_ptr, key_ptr);
        let tensor = NslTensor::from_ptr(tensor_ptr);

        let shape: Vec<usize> = (0..tensor.ndim as usize)
            .map(|d| unsafe { *tensor.shape.add(d) as usize })
            .collect();

        // Convert to f32
        let f32_vec: Vec<f32> = if tensor.dtype == 1 {
            // Already f32
            let slice = unsafe {
                std::slice::from_raw_parts(tensor.data_f32(), tensor.len as usize)
            };
            slice.to_vec()
        } else {
            // f64 -> f32
            let slice = unsafe {
                std::slice::from_raw_parts(tensor.data_f64(), tensor.len as usize)
            };
            slice.iter().map(|&v| v as f32).collect()
        };

        tensors_data.push((key_str, f32_vec, shape));
    }

    // Build TensorView references
    let tensor_views: Vec<(&str, safetensors::tensor::TensorView<'_>)> = tensors_data
        .iter()
        .map(|(name, data, shape)| {
            let view = safetensors::tensor::TensorView::new(
                safetensors::Dtype::F32,
                shape,
                bytemuck::cast_slice(data),
            ).unwrap();
            (name.as_str(), view)
        })
        .collect();

    if let Err(e) = safetensors::serialize_to_file(&tensor_views, &None, std::path::Path::new(path)) {
        eprintln!("[nsl] Error writing safetensors file '{}': {}", path, e);
        std::process::abort();
    }
}
```

```rust
/// Map a HuggingFace parameter name to an NSL model field path.
/// Applies standard substitutions, then user overrides.
pub fn map_hf_name(hf_name: &str, overrides: &HashMap<String, String>) -> String {
    // Check user overrides first
    if let Some(mapped) = overrides.get(hf_name) {
        return mapped.clone();
    }

    let mut name = hf_name.to_string();

    // Apply standard substitutions
    for &(from, to) in NAME_MAPPINGS {
        name = name.replace(from, to);
    }

    // Convert "layers.N." to "layers[N]."
    let re_pattern = regex_lite::Regex::new(r"layers\.(\d+)\.").unwrap();

pub fn needs_transpose(field_name: &str, layer_type: &str) -> bool {

/// Validate that all model params have matching weights, and all weights are consumed.
/// Returns (matched_pairs, unmapped_params, unused_weights).
pub fn validate_mapping(
    model_params: &[String],
    weight_names: &[String],
    overrides: &HashMap<String, String>,
) -> (Vec<(String, String)>, Vec<String>, Vec<String>) {
    let mut matched = Vec::new();
    let mut unused: Vec<String> = weight_names.to_vec();

    let mapped_weights: HashMap<String, String> = weight_names
        .iter()
        .map(|w| (map_hf_name(w, overrides), w.clone()))
        .collect();

    let mut unmapped = Vec::new();

    for param in model_params {
        if let Some(hf_name) = mapped_weights.get(param) {
            matched.push((param.clone(), hf_name.clone()));
            unused.retain(|w| w != hf_name);
        } else {
            unmapped.push(param.clone());
        }
    }

    (matched, unmapped, unused)
}
```

```rust
/// Load weights from an NslDict into a model struct using the offset table.
/// This is the core logic, separated from HF download for testability.
pub fn load_weights_from_dict(model_ptr: i64, metadata: &[ParamMeta], dict_ptr: i64) {
    let keys_list_ptr = crate::dict::nsl_dict_keys(dict_ptr);
    let keys_list = crate::list::NslList::from_ptr(keys_list_ptr);

    // Build a map of available weight names
    let mut available: std::collections::HashMap<String, i64> = std::collections::HashMap::new();
    for i in 0..keys_list.len as usize {
        let key_ptr = unsafe { *keys_list.data.add(i) };
        let key_cstr = unsafe { CStr::from_ptr(key_ptr as *const c_char) };
        let key = key_cstr.to_str().unwrap_or("").to_string();
        let val = crate::dict::nsl_dict_get_str(dict_ptr, key_ptr);
        available.insert(key, val);
    }

    let mut matched_count = 0;

    for meta in metadata {
        let param_name = unsafe { CStr::from_ptr(meta.name_ptr) }
            .to_str()
            .unwrap_or("");

        // Find matching weight
        let tensor_ptr = if let Some(&ptr) = available.get(param_name) {
            ptr
        } else {
            eprintln!("[nsl] Error: no weight found for model parameter '{}'", param_name);
            std::process::abort();
        };

        let tensor = NslTensor::from_ptr(tensor_ptr);

        // Validate shape
        let expected_shape: Vec<i64> = (0..meta.ndim as usize)
            .map(|d| unsafe { *meta.shape_ptr.add(d) })
            .collect();
        let actual_shape: Vec<i64> = (0..tensor.ndim as usize)
            .map(|d| unsafe { *tensor.shape.add(d) })
            .collect();

        // If transpose is needed, expected shape is reversed
        let check_shape = if meta.transpose != 0 {
            vec![expected_shape[1], expected_shape[0]]
        } else {
            expected_shape.clone()
        };

        if actual_shape != check_shape {
            eprintln!(
                "[nsl] Shape mismatch for '{}': expected {:?}, got {:?}",
                param_name, expected_shape, actual_shape
            );
            std::process::abort();
        }

        // Transpose if needed
        let final_tensor_ptr = if meta.transpose != 0 && tensor.ndim == 2 {
            transpose_2d(tensor_ptr)
        } else {
            // Clone the tensor so model owns its own copy
            crate::tensor::nsl_tensor_clone(tensor_ptr)
        };

        // Write tensor pointer into model struct at the correct offset
        unsafe {
            let dest = (model_ptr as *mut u8).add(meta.offset as usize) as *mut i64;
            *dest = final_tensor_ptr;
        }

        matched_count += 1;
    }

    if matched_count != metadata.len() {
        eprintln!(
            "[nsl] Warning: only {}/{} parameters mapped",
            matched_count,
            metadata.len()
        );
    }

    // Warn about unused weights
    let used_names: std::collections::HashSet<String> = metadata
        .iter()
        .map(|m| unsafe { CStr::from_ptr(m.name_ptr) }.to_str().unwrap_or("").to_string())
        .collect();
    for name in available.keys() {
        if !used_names.contains(name) {
            eprintln!("[nsl] Warning: unused weight '{}'", name);
        }
    }
}

/// Transpose a 2D f32 tensor [M, N] -> [N, M].
fn transpose_2d(tensor_ptr: i64) -> i64 {
    let src = NslTensor::from_ptr(tensor_ptr);
    assert_eq!(src.ndim, 2, "transpose_2d requires 2D tensor");
    assert_eq!(src.dtype, 1, "transpose_2d requires f32 tensor");

    let m = unsafe { *src.shape } as usize;
    let n = unsafe { *src.shape.add(1) } as usize;

    let src_data = src.data_f32();
    let dst_bytes = m * n * std::mem::size_of::<f32>();
    let dst_data = unsafe { crate::memory::checked_alloc(dst_bytes) } as *mut f32;

    // Transpose: dst[j * m + i] = src[i * n + j]
    for i in 0..m {
        for j in 0..n {
            unsafe {
                *dst_data.add(j * m + i) = *src_data.add(i * n + j);
            }
        }
    }

    let shape_ptr = unsafe { crate::memory::checked_alloc(16) } as *mut i64;
    unsafe {
        *shape_ptr = n as i64;
        *shape_ptr.add(1) = m as i64;
    }
    let strides_ptr = NslTensor::compute_strides(shape_ptr, 2);

    let result = NslTensor {
        data: dst_data as *mut c_void,
        shape: shape_ptr,
        strides: strides_ptr,
        ndim: 2,
        len: (m * n) as i64,
        refcount: 1,
        device: src.device,
        dtype: 1,
    };

    Box::into_raw(Box::new(result)) as i64
}
```

```rust
/// Called by compiler-generated code with the static ParamMeta offset table.
#[no_mangle]
pub extern "C" fn nsl_hf_load(
    model_ptr: i64,
    metadata_ptr: i64,
    metadata_len: i64,
    repo_id_ptr: i64,
    repo_id_len: i64,
    device: i64,
) -> i64 {
    let repo_id = unsafe {
        let slice = std::slice::from_raw_parts(repo_id_ptr as *const u8, repo_id_len as usize);
        std::str::from_utf8_unchecked(slice)
    };

    let metadata = unsafe {
        std::slice::from_raw_parts(
            metadata_ptr as *const ParamMeta,
            metadata_len as usize,
        )
    };

    // Download safetensors files from HF Hub
    let api = hf_hub::api::sync::Api::new().unwrap_or_else(|e| {
        eprintln!("[nsl] Failed to initialize HF Hub API: {}", e);
        std::process::abort();
    });

    let repo = api.model(repo_id.to_string());

    // Try to download model.safetensors first, then fall back to sharded
    let safetensors_path = match repo.get("model.safetensors") {
        Ok(path) => path,
        Err(_) => {
            // Try safetensors index for sharded models
            match repo.get("model.safetensors.index.json") {
                Ok(index_path) => {
                    // Parse index to get shard filenames
                    let index_content = std::fs::read_to_string(&index_path).unwrap_or_else(|e| {
                        eprintln!("[nsl] Failed to read safetensors index: {}", e);
                        std::process::abort();
                    });
                    // For now, just use the first shard
                    // TODO: handle multi-shard models properly
                    eprintln!("[nsl] Multi-shard models not yet supported");
                    std::process::abort();
                }
                Err(e) => {
                    eprintln!("[nsl] Failed to download model from '{}': {}", repo_id, e);
                    std::process::abort();
                }
            }
        }
    };

    // Load safetensors into dict
    let path_str = safetensors_path.to_str().unwrap();
    let dict_ptr = crate::safetensors_io::nsl_safetensors_load(
        path_str.as_ptr() as i64,
        path_str.len() as i64,
        device,
    );

    // Map weights into model struct
    load_weights_from_dict(model_ptr, metadata, dict_ptr);

    0 // success
}
```

```rust
/// Generate ParamMeta entries for a model type by recursively walking its field tree.
/// Returns (entries, total_metadata_size_bytes).
/// Each entry: (name_string, byte_offset, shape_dims, ndim, needs_transpose)
fn generate_param_metadata(
    &self,
    model_name: &str,
    prefix: &str,
    base_offset: usize,
) -> Vec<(String, usize, Vec<i64>, bool)> {
    let mut entries = Vec::new();

    let layout = match self.struct_layouts.get(model_name) {
        Some(l) => l.clone(),
        None => return entries,
    };

    let field_types = self.model_field_types.get(model_name).cloned().unwrap_or_default();

    for field in &layout.fields {
        let field_path = if prefix.is_empty() {
            field.name.clone()
        } else {
            format!("{}.{}", prefix, field.name)
        };

        let abs_offset = base_offset + field.offset;

        // Check if this field is a nested model, an array, or a leaf tensor
        if let Some(type_marker) = field_types.get(&field.name) {
            if type_marker.starts_with('[') {
                // FixedModelArray: [LayerType;N]
                // Parse "[LayerType;N]"
                let inner = &type_marker[1..type_marker.len() - 1];
                let parts: Vec<&str> = inner.split(';').collect();
                if parts.len() == 2 {
                    let elem_type = parts[0];
                    let count: usize = parts[1].parse().unwrap_or(0);
                    let elem_layout = self.struct_layouts.get(elem_type);
                    let elem_size = elem_layout.map(|l| l.total_size).unwrap_or(8);

                    for i in 0..count {
                        let array_prefix = format!("{}[{}]", field_path, i);
                        let array_offset = abs_offset + i * elem_size;
                        let sub_entries = self.generate_param_metadata(
                            elem_type,
                            &array_prefix,
                            array_offset,
                        );
                        entries.extend(sub_entries);
                    }
                }
            } else {
                // Nested model: recurse
                let sub_entries = self.generate_param_metadata(
                    type_marker,
                    &field_path,
                    abs_offset,
                );
                entries.extend(sub_entries);
            }
        } else {
            // Leaf field (tensor parameter)
            // Determine transpose based on field name and parent model type
            let needs_transpose = is_linear_weight(&field.name, model_name, &self.struct_layouts);

            // Shape is unknown at compile time — pass empty, runtime validates
            entries.push((field_path, abs_offset, vec![], needs_transpose));
        }
    }

    entries
}
```

```rust
/// Compile a from_hf("repo_id", model, device="cpu") call.
/// Generates the ParamMeta offset table in .rodata and emits nsl_hf_load.
fn compile_from_hf(
    &mut self,
    builder: &mut FunctionBuilder,
    state: &mut FuncState,
    args: &[nsl_ast::expr::Arg],
) -> Result<Value, CodegenError> {
    // arg 0: repo_id (string literal)
    let repo_id_val = self.compile_expr(builder, state, &args[0].value)?;
    // arg 1: model variable
    let model_val = self.compile_expr(builder, state, &args[1].value)?;

    // Resolve model type name
    let model_type = self.node_type(args[1].value.id).clone();
    let model_name = match &model_type {
        nsl_ast::types::Type::Model { name, .. } => self.resolve_sym(*name).to_string(),
        _ => return Err(CodegenError::new("from_hf: second argument must be a model".into())),
    };

    // Get device argument (default 0 = CPU)
    let device_val = if args.len() > 2 {
        self.compile_expr(builder, state, &args[2].value)?
    } else {
        builder.ins().iconst(cranelift_codegen::ir::types::I64, 0)
    };

    // Generate ParamMeta entries
    let metadata = self.generate_param_metadata(&model_name, "", 0);

    if metadata.is_empty() {
        return Err(CodegenError::new(format!(
            "from_hf: model '{}' has no loadable parameters", model_name
        )));
    }

    let meta_count = metadata.len();

    // Emit each ParamMeta as static data in .rodata
    // Layout: [name_ptr(8), offset(8), shape_ptr(8), ndim(8), transpose(8)] × N
    let meta_entry_size = 40; // 5 × 8 bytes
    let meta_total = meta_count * meta_entry_size;

    // Allocate metadata array at runtime (simpler than .rodata for variable-length data)
    let meta_size_val = builder.ins().iconst(cranelift_codegen::ir::types::I64, meta_total as i64);
    let meta_array = self.compile_call_by_name(builder, "nsl_alloc", &[meta_size_val])?;

    for (i, (name, offset, _shape, transpose)) in metadata.iter().enumerate() {
        let entry_offset = (i * meta_entry_size) as i64;

        // Store name string pointer
        let name_data_id = self.intern_string(name)?;
        let name_gv = self.module.declare_data_in_func(name_data_id, builder.func);
        let name_ptr = builder.ins().symbol_value(cranelift_codegen::ir::types::I64, name_gv);
        builder.ins().store(
            cranelift_codegen::ir::MemFlags::trusted(),
            name_ptr,
            meta_array,
            cranelift_codegen::ir::immediates::Offset32::new(entry_offset as i32),
        );

        // Store offset
        let offset_val = builder.ins().iconst(cranelift_codegen::ir::types::I64, *offset as i64);
        builder.ins().store(
            cranelift_codegen::ir::MemFlags::trusted(),
            offset_val,
            meta_array,
            cranelift_codegen::ir::immediates::Offset32::new((entry_offset + 8) as i32),
        );

        // Store shape_ptr (0 = no compile-time shape, runtime validates)
        let shape_null = builder.ins().iconst(cranelift_codegen::ir::types::I64, 0);
        builder.ins().store(
            cranelift_codegen::ir::MemFlags::trusted(),
            shape_null,
            meta_array,
            cranelift_codegen::ir::immediates::Offset32::new((entry_offset + 16) as i32),
        );

        // Store ndim (0 = unknown)
        let ndim_val = builder.ins().iconst(cranelift_codegen::ir::types::I64, 0);
        builder.ins().store(
            cranelift_codegen::ir::MemFlags::trusted(),
            ndim_val,
            meta_array,
            cranelift_codegen::ir::immediates::Offset32::new((entry_offset + 24) as i32),
        );

        // Store transpose flag
        let transpose_val = builder.ins().iconst(
            cranelift_codegen::ir::types::I64,
            if *transpose { 1 } else { 0 },
        );
        builder.ins().store(
            cranelift_codegen::ir::MemFlags::trusted(),
            transpose_val,
            meta_array,
            cranelift_codegen::ir::immediates::Offset32::new((entry_offset + 32) as i32),
        );
    }

    let meta_len = builder.ins().iconst(cranelift_codegen::ir::types::I64, meta_count as i64);

    // Get repo_id string length
    let repo_id_len = self.compile_call_by_name(builder, "nsl_string_len", &[repo_id_val])?;

    // Call nsl_hf_load(model_ptr, meta_ptr, meta_len, repo_id_ptr, repo_id_len, device)
    let _result = self.compile_call_by_name(
        builder,
        "nsl_hf_load",
        &[model_val, meta_array, meta_len, repo_id_val, repo_id_len, device_val],
    )?;

    // Return the model pointer (modified in-place)
    Ok(model_val)
}
```

```rust
/// Start tracing mode. Allocates a new TraceGraph.
#[no_mangle]
pub extern "C" fn nsl_trace_start() {
    let mut guard = TRACE_GRAPH.lock().unwrap();
    *guard = Some(TraceGraph::new());
    TRACING.store(true, Ordering::SeqCst);
}

/// Register a tensor pointer as a graph input.
#[no_mangle]
pub extern "C" fn nsl_trace_register_input(tensor_ptr: i64, name_ptr: i64) {
    let name = unsafe {
        std::ffi::CStr::from_ptr(name_ptr as *const i8)
            .to_str()
            .unwrap_or("input")
            .to_string()
    };
    let mut guard = TRACE_GRAPH.lock().unwrap();
    if let Some(graph) = guard.as_mut() {
        graph.inputs.push((tensor_ptr, name));
    }
}

/// Register a tensor pointer as a graph output.
#[no_mangle]
pub extern "C" fn nsl_trace_register_output(tensor_ptr: i64, name_ptr: i64) {
    let name = unsafe {
        std::ffi::CStr::from_ptr(name_ptr as *const i8)
            .to_str()
            .unwrap_or("output")
            .to_string()
    };
    let mut guard = TRACE_GRAPH.lock().unwrap();
    if let Some(graph) = guard.as_mut() {
        graph.outputs.push((tensor_ptr, name));
    }
}

/// Stop tracing and return a pointer to the completed TraceGraph.
#[no_mangle]
pub extern "C" fn nsl_trace_stop() -> i64 {
    TRACING.store(false, Ordering::SeqCst);
    let mut guard = TRACE_GRAPH.lock().unwrap();
    match guard.take() {
        Some(graph) => Box::into_raw(Box::new(graph)) as i64,
        None => 0,
    }
}

/// Record a tensor operation in the trace. Called by runtime ops when TRACING is active.
pub fn record_op(
    op_type: OpType,
    input_ptrs: &[i64],
    output_ptr: i64,
    output_shape: &[i64],
    output_dtype: u8,
    attributes: &[(String, AttrValue)],
) {
    let mut guard = TRACE_GRAPH.lock().unwrap();
    if let Some(graph) = guard.as_mut() {
        let node_id = graph.ops.len();
        graph.ops.push(TraceOp {
            op_type,
            input_ptrs: input_ptrs.to_vec(),
            output_ptr,
            output_shape: output_shape.to_vec(),
            output_dtype,
            attributes: attributes.to_vec(),
        });
        graph.ptr_to_node.insert(output_ptr, node_id);
    }
}

/// Check if tracing is active. Called by runtime ops before recording.
#[inline]
pub fn is_tracing() -> bool {
    TRACING.load(Ordering::Relaxed)
}
```

```rust
/// Resolve a tensor pointer to its ONNX edge name.
/// Returns (name, is_initializer).
fn resolve_ptr(
    ptr: i64,
    graph: &TraceGraph,
    edge_names: &mut HashMap<i64, String>,
    initializer_counter: &mut usize,
) -> (String, bool) {
    // Already resolved?
    if let Some(name) = edge_names.get(&ptr) {
        return (name.clone(), false);
    }

    // Is it a registered input?
    for (p, name) in &graph.inputs {
        if *p == ptr {
            edge_names.insert(ptr, name.clone());
            return (name.clone(), false);
        }
    }

    // Is it the output of a prior op?
    if let Some(&node_id) = graph.ptr_to_node.get(&ptr) {
        let name = format!("node_{}_out", node_id);
        edge_names.insert(ptr, name.clone());
        return (name, false);
    }

    // Neither — it's a static weight/constant (initializer)
    let name = format!("initializer_{}", *initializer_counter);
    *initializer_counter += 1;
    edge_names.insert(ptr, name.clone());
    (name, true)
}

/// Build an ONNX ModelProto from a TraceGraph.

    let mut edge_names: HashMap<i64, String> = HashMap::new();
    let mut init_counter = 0;
    let mut nodes = Vec::new();
    let mut initializers: Vec<(String, i64)> = Vec::new(); // (name, tensor_ptr)

    // Pre-register input names
    for (ptr, name) in &graph.inputs {
        edge_names.insert(*ptr, name.clone());
    }

    // Build nodes
    for (i, op) in graph.ops.iter().enumerate() {
        let mut input_names = Vec::new();
        for &iptr in &op.input_ptrs {
            let (name, is_init) = resolve_ptr(iptr, graph, &mut edge_names, &mut init_counter);
            if is_init {
                initializers.push((name.clone(), iptr));
            }
            input_names.push(name);
        }

        let output_name = format!("node_{}_out", i);
        edge_names.insert(op.output_ptr, output_name.clone());

        // Convert attributes
        let onnx_attrs: Vec<AttributeProto> = op.attributes.iter().map(|(name, val)| {
            match val {
                AttrValue::Int(v) => AttributeProto {
                    name: name.clone(),
                    r#type: ATTRIBUTE_INT,
                    i: *v,
                    ..Default::default()
                },
                AttrValue::Float(v) => AttributeProto {
                    name: name.clone(),
                    r#type: ATTRIBUTE_FLOAT,
                    f: *v as f32,
                    ..Default::default()
                },
                AttrValue::Ints(v) => AttributeProto {
                    name: name.clone(),
                    r#type: ATTRIBUTE_INTS,
                    ints: v.clone(),
                    ..Default::default()
                },
            }
        }).collect();

        nodes.push(NodeProto {
            input: input_names,
            output: vec![output_name],
            name: format!("{}_{}", op_type_to_onnx(&op.op_type).to_lowercase(), i),
            op_type: op_type_to_onnx(&op.op_type).to_string(),
            domain: "".to_string(),
            attribute: onnx_attrs,
        });
    }

    // Remap output names
    for (ptr, name) in &graph.outputs {
        if let Some(edge_name) = edge_names.get(ptr) {
            // Find the last node that produces this output and rename it
            for node in nodes.iter_mut().rev() {
                if node.output.contains(edge_name) {
                    node.output = vec![name.clone()];
                    break;
                }
            }
        }
    }

    // Build initializer TensorProtos (read actual data from memory)
    let onnx_initializers: Vec<TensorProto> = initializers
        .iter()
        .filter_map(|(name, ptr)| {
            if *ptr == 0 {
                return None;
            }
            // Safety: only valid during tracing when tensors are live
            let tensor = NslTensor::from_ptr(*ptr);
            let shape: Vec<i64> = (0..tensor.ndim as usize)
                .map(|d| unsafe { *tensor.shape.add(d) })
                .collect();

            let (data_type, raw_data) = if tensor.dtype == 1 {
                let data = unsafe {
                    std::slice::from_raw_parts(
                        tensor.data_f32() as *const u8,
                        tensor.len as usize * 4,
                    )
                };
                (FLOAT, data.to_vec())
            } else {
                let data = unsafe {
                    std::slice::from_raw_parts(
                        tensor.data_f64() as *const u8,
                        tensor.len as usize * 8,
                    )
                };
                (DOUBLE, data.to_vec())
            };

            Some(TensorProto {
                dims: shape,
                data_type,
                name: name.clone(),
                raw_data,
            })
        })
        .collect();

    // Build input ValueInfoProtos
    let onnx_inputs: Vec<ValueInfoProto> = graph
        .inputs
        .iter()
        .map(|(ptr, name)| {
            // Try to get shape from the tensor
            let (elem_type, dims) = if *ptr != 0 {
                let tensor = NslTensor::from_ptr(*ptr);
                let shape: Vec<i64> = (0..tensor.ndim as usize)
                    .map(|d| unsafe { *tensor.shape.add(d) })
                    .collect();
                let dt = if tensor.dtype == 1 { FLOAT } else { DOUBLE };
                (dt, shape)
            } else {
                (FLOAT, vec![])
            };

            ValueInfoProto {
                name: name.clone(),
                r#type: Some(TypeProto {
                    tensor_type: Some(TensorTypeProto {
                        elem_type,
                        shape: Some(TensorShapeProto {
                            dim: dims.iter().map(|&d| TensorShapeDimProto { dim_value: d }).collect(),
                        }),
                    }),
                }),
            }
        })
        .collect();

    // Build output ValueInfoProtos
    let onnx_outputs: Vec<ValueInfoProto> = graph
        .outputs
        .iter()
        .map(|(ptr, name)| {
            let (elem_type, dims) = if *ptr != 0 {
                let tensor = NslTensor::from_ptr(*ptr);
                let shape: Vec<i64> = (0..tensor.ndim as usize)
                    .map(|d| unsafe { *tensor.shape.add(d) })
                    .collect();
                let dt = if tensor.dtype == 1 { FLOAT } else { DOUBLE };
                (dt, shape)
            } else {
                (FLOAT, vec![])
            };

            ValueInfoProto {
                name: name.clone(),
                r#type: Some(TypeProto {
                    tensor_type: Some(TensorTypeProto {
                        elem_type,
                        shape: Some(TensorShapeProto {
                            dim: dims.iter().map(|&d| TensorShapeDimProto { dim_value: d }).collect(),
                        }),
                    }),
                }),
            }
        })
        .collect();

    ModelProto {
        ir_version: 8,
        producer_name: "nsl".to_string(),
        producer_version: "0.1".to_string(),
        opset_import: vec![OperatorSetIdProto {
            domain: "".to_string(),
            version: 17,
        }],
        graph: Some(GraphProto {
            name: "nsl_model".to_string(),
            node: nodes,
            initializer: onnx_initializers,
            input: onnx_inputs,
            output: onnx_outputs,
        }),
    }
}

/// Export a TraceGraph to an ONNX file.
#[no_mangle]
pub extern "C" fn nsl_onnx_export(trace_ptr: i64, path_ptr: i64, path_len: i64) {
    let graph = unsafe { &*(trace_ptr as *const TraceGraph) };
    let path = unsafe {
        let slice = std::slice::from_raw_parts(path_ptr as *const u8, path_len as usize);
        std::str::from_utf8_unchecked(slice)
    };

    let model = build_onnx_model(graph);

    // Serialize to protobuf
    use prost::Message;
    let mut buf = Vec::new();
    if let Err(e) = model.encode(&mut buf) {
        eprintln!("[nsl] Failed to encode ONNX model: {}", e);
        std::process::abort();
    }

    // Write to file
    if let Err(e) = std::fs::write(path, &buf) {
        eprintln!("[nsl] Failed to write ONNX file '{}': {}", path, e);
        std::process::abort();
    }

    eprintln!("[nsl] Exported ONNX model to '{}' ({} bytes, {} nodes)",
        path, buf.len(), model.graph.map(|g| g.node.len()).unwrap_or(0));
}
```

```rust
/// Compile a to_onnx(model, input, "output.onnx") call.
/// Emits: trace_start, register_input, model.forward(input), register_output, trace_stop, export.
fn compile_to_onnx(
    &mut self,
    builder: &mut FunctionBuilder,
    state: &mut FuncState,
    args: &[nsl_ast::expr::Arg],
) -> Result<Value, CodegenError> {
    if args.len() < 3 {
        return Err(CodegenError::new("to_onnx requires 3 arguments: (model, input, path)".into()));
    }

    // Compile arguments
    let model_val = self.compile_expr(builder, state, &args[0].value)?;
    let input_val = self.compile_expr(builder, state, &args[1].value)?;
    let path_val = self.compile_expr(builder, state, &args[2].value)?;

    // Resolve model type to find its forward method
    let model_type = self.node_type(args[0].value.id).clone();
    let model_name = match &model_type {
        nsl_ast::types::Type::Model { name, .. } => self.resolve_sym(*name).to_string(),
        _ => return Err(CodegenError::new("to_onnx: first argument must be a model".into())),
    };

    // 1. nsl_trace_start()
    self.compile_call_by_name(builder, "nsl_trace_start", &[])?;

    // 2. nsl_trace_register_input(input_ptr, "input_0")
    let input_name_id = self.intern_string("input_0")?;
    let input_name_gv = self.module.declare_data_in_func(input_name_id, builder.func);
    let input_name_ptr = builder.ins().symbol_value(cranelift_codegen::ir::types::I64, input_name_gv);
    self.compile_call_by_name(builder, "nsl_trace_register_input", &[input_val, input_name_ptr])?;

    // 3. Call model.forward(input)
    let forward_name = format!("{}_forward", model_name);
    let result = self.compile_call_by_name(builder, &forward_name, &[model_val, input_val])?;

    // 4. nsl_trace_register_output(result, "output_0")
    let output_name_id = self.intern_string("output_0")?;
    let output_name_gv = self.module.declare_data_in_func(output_name_id, builder.func);
    let output_name_ptr = builder.ins().symbol_value(cranelift_codegen::ir::types::I64, output_name_gv);
    self.compile_call_by_name(builder, "nsl_trace_register_output", &[result, output_name_ptr])?;

    // 5. let trace = nsl_trace_stop()
    let trace = self.compile_call_by_name(builder, "nsl_trace_stop", &[])?;

    // 6. nsl_onnx_export(trace, path_ptr, path_len)
    let path_len = self.compile_call_by_name(builder, "nsl_string_len", &[path_val])?;
    self.compile_call_by_name(builder, "nsl_onnx_export", &[trace, path_val, path_len])?;

    Ok(result)
}
```
